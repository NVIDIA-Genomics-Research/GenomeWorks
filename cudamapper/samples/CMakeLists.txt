get_property(cudamapper_data_include_dir GLOBAL PROPERTY cudamapper_data_include_dir)

               sample_cudamapper.cpp
               )

                      cudamapper

    target_compile_definitions(${MODULE_NAME}
                               PUBLIC
                               GW_BUILD_HTSLIB)
endif()

            DESTINATION samples)
